sqlite를 이용해 2명의 이름과 나이를 생성하고db에서 읽고 화면에 출력
```
#include <stdio.h>
#include <sqlite3.h>

int print_callback(void* data, int argc, char** argv, char** colName)
{
    printf("이름: %s, 나이: %s\n", argv[0], argv[1]);
    return 0;
}

int main(void)
{
    sqlite3* db;
    char* errMsg = 0;

    sqlite3_open("student.db", &db);

    sqlite3_exec(db,
        "CREATE TABLE IF NOT EXISTS student ("
        "name TEXT, "
        "age INTEGER);",
        0, 0, &errMsg);

    sqlite3_exec(db, "DELETE FROM student;", 0, 0, &errMsg);

    sqlite3_exec(db,
        "INSERT INTO student VALUES ('Kim', 20);"
        "INSERT INTO student VALUES ('Lee', 22);",
        0, 0, &errMsg);

    printf("===== DB에서 읽은 학생 정보 =====\n");

    sqlite3_exec(db,
        "SELECT name, age FROM student;",
        print_callback, 0, &errMsg);

    sqlite3_close(db);

    return 0;
}
```
