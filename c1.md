무지개 블록꺠기
raylib를 nuget에서 설치한다.
```
#include "raylib.h"

#define W 800
#define H 600

#define ROWS 6
#define COLS 10

int main(void)
{
    InitWindow(W, H, "Rainbow Brick Breaker");
    SetTargetFPS(60);

    Rectangle paddle;
    Vector2 ball;
    Vector2 speed;

    Rectangle bricks[ROWS][COLS];
    int alive[ROWS][COLS];

    Color colors[ROWS] = {
        RED, ORANGE, YELLOW, GREEN, BLUE, PURPLE
    };

    int score = 0;
    int lives = 3;
    int gameOver = 0;
    int gameClear = 0;
    int start = 0;

restart:

    paddle = (Rectangle){ W / 2 - 50, H - 50, 100, 15 };
    ball = (Vector2){ W / 2, H - 70 };
    speed = (Vector2){ 4, -4 };

    score = 0;
    lives = 3;
    gameOver = 0;
    gameClear = 0;
    start = 0;

    for (int y = 0; y < ROWS; y++) {
        for (int x = 0; x < COLS; x++) {
            bricks[y][x] = (Rectangle){
                40 + x * 72,
                60 + y * 30,
                65,
                22
            };
            alive[y][x] = 1;
        }
    }

    while (!WindowShouldClose()) {
        if (!gameOver && !gameClear) {
            if (IsKeyDown(KEY_LEFT) || IsKeyDown(KEY_A))
                paddle.x -= 6;

            if (IsKeyDown(KEY_RIGHT) || IsKeyDown(KEY_D))
                paddle.x += 6;

            if (paddle.x < 0) paddle.x = 0;
            if (paddle.x + paddle.width > W) paddle.x = W - paddle.width;

            if (!start) {
                ball.x = paddle.x + paddle.width / 2;
                ball.y = paddle.y - 10;

                if (IsKeyPressed(KEY_SPACE))
                    start = 1;
            }
            else {
                ball.x += speed.x;
                ball.y += speed.y;
            }

            if (ball.x < 8 || ball.x > W - 8)
                speed.x *= -1;

            if (ball.y < 8)
                speed.y *= -1;

            if (CheckCollisionCircleRec(ball, 8, paddle)) {
                speed.y *= -1;
                ball.y = paddle.y - 10;
            }

            for (int y = 0; y < ROWS; y++) {
                for (int x = 0; x < COLS; x++) {
                    if (alive[y][x] &&
                        CheckCollisionCircleRec(ball, 8, bricks[y][x])) {
                        alive[y][x] = 0;
                        speed.y *= -1;
                        score += 10;
                    }
                }
            }

            if (ball.y > H) {
                lives--;

                if (lives <= 0) {
                    gameOver = 1;
                }
                else {
                    start = 0;
                    ball.x = paddle.x + paddle.width / 2;
                    ball.y = paddle.y - 10;
                    speed = (Vector2){ 4, -4 };
                }
            }

            if (score == ROWS * COLS * 10)
                gameClear = 1;
        }

        if ((gameOver || gameClear) && IsKeyPressed(KEY_R))
            goto restart;

        BeginDrawing();
        ClearBackground(BLACK);

        for (int y = 0; y < ROWS; y++) {
            for (int x = 0; x < COLS; x++) {
                if (alive[y][x])
                    DrawRectangleRec(bricks[y][x], colors[y]);
            }
        }

        DrawRectangleRec(paddle, WHITE);
        DrawCircleV(ball, 8, WHITE);

        DrawText(TextFormat("Score: %d", score), 20, 20, 20, WHITE);
        DrawText(TextFormat("Lives: %d", lives), 680, 20, 20, WHITE);

        if (!start && !gameOver && !gameClear)
            DrawText("Press SPACE", 330, 530, 24, WHITE);

        if (gameOver) {
            DrawText("GAME OVER", 290, 270, 40, RED);
            DrawText("Press R to Restart", 285, 320, 24, WHITE);
        }

        if (gameClear) {
            DrawText("CLEAR!", 330, 270, 40, YELLOW);
            DrawText("Press R to Restart", 285, 320, 24, WHITE);
        }

        EndDrawing();
    }

    CloseWindow();
    return 0;
}
```
